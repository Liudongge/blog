# 在浏览器中预览文档，标签名称自定义方案

# 方案一、延时设置newWindow.document.title

图片预览，直接延时判断title是否存在并设置newWindow.document.title即可

const newWindow = window.open(`${window.location.origin}${process.env.VUE_APP_BASE_API}'/clientplatform/file/downloadImage?id='${documentId}`, '_blank')
let countDownTime = 20
let timerId = setInterval(
  () => {
    if ((newWindow.document && newWindow.document.title.includes('downloadImage'))) {
      newWindow.document.title = fileName || '文件名称-title'
      clearInterval(timerId)
      timerId = null
    }
  },
  100
)

PDF预览时，获取到的newWindow.document.title一直为空，无法进行有效判断，考虑重复设置

const newWindow = window.open(`${window.location.origin}${process.env.VUE_APP_BASE_API}/clientplatform/file/downloadImage?id=${documentId}`, '_blank')
newWindow.document.title = fileName || 'test'
let countDownTime = 20
let timerId = setInterval(
  () => {
    if ((newWindow.document && newWindow.document.title.includes('downloadImage')) || countDownTime <= 0) {
      newWindow.document.title = fileName || '预览'
      clearInterval(timerId)
      timerId = null
    } else {
      countDownTime--
      // hack：十秒内重复设置标题，解决文件过大加载缓慢后downloadImage再次覆盖的问题
      newWindow.document.title = fileName || '预览'
    }
  },
  500
)

# 方案二、iframe

iframe加载文件，无需延时直接设置newWindow.document.title

// 在新tab中使用iframe预览pdf文件，设置标签名称
const newWindow = window.open('', '_blank')
newWindow.document.write(
  `<iframe src="${window.location.origin}${process.env.VUE_APP_BASE_API}/clientplatform/file/downloadImage?id=${documentId}" width="100%" height="100%" allowfullscreen></iframe>`
)
newWindow.document.title = fileName || '预览'
