# HTTPS

问题描述：测试环境使用HTTPS访问，上报日志时出现浏览器拦截的情况。

解决方案：

  • 去除请求头，直接使用 //xxx.xxx的访问方式。原理是使用域名下默认的请求头，解决浏览器对https域名下请求http报错的问题，但是仍然会出现ERR_SSL_PROTOCOL_ERR 的错误 。 ❎
  • 日志服务使用HTTPS部署。✅

解决步骤：

  1. 通过mkcert创建CA证书。搭建https环境。

  2. 通过nginx配置代理webfunny的web和日志服务。 nginx配置如下：

    server {
  listen 80;
  server_name x.x.x.x;
  root /app;
  index /home.html;

  location /webfunny/ {
    proxy_pass http://x.x.x.x:webport/webfunny/;
    proxy_redirect default ;
  }
  # webfunny_server 数据接口等反向代理
  location /server/ {
    proxy_pass http://x.x.x.x:serverport/server/;
    proxy_connect_timeout 3000;
    proxy_send_timeout 3000;
    proxy_set_header Host $http_host;
    proxy_set_header X-Real-IP $remote_addr;
    proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    proxy_set_header X-Forwarded-Proto $scheme;
  }
}

# localServerDomain: 'x.x.x.x:serverport', // 日志服务域名  书写形式：localhost:8011
# localAssetsDomain: 'x.x.x.x:webport', // 数据可视化服务域名 书写形式：localhost:8010
# localServerPort: '8011', // 日志服务端口号
# localAssetsPort: '8010', // 可视化系统端口号
server {
  listen       443 ssl;
  server_name  x.x.x.x;

  ssl_certificate      /app/nginx/x.x.x.x.pem;
  ssl_certificate_key  /app/nginx/x.x.x.x-key.pem;
  ssl_session_cache    shared:SSL:1m;
  ssl_session_timeout  5m;
  # 指定密码为openssl支持的格式
  ssl_protocols  SSLv2 SSLv3 TLSv1.2;
  ssl_ciphers  HIGH:!aNULL:!MD5;  # 密码加密方式
  ssl_prefer_server_ciphers  on;   # 依赖SSLv3和TLSv1协议的服务器密码将优先于客户端密码
  
  root /app;
  index /index.html;
  # 数据接口等代理
  location /server/ {
    proxy_pass http://x.x.x.x:serverport/server/;
    proxy_connect_timeout 3000;
    proxy_send_timeout 3000;
    proxy_set_header Host $http_host;
    proxy_set_header X-Real-IP $remote_addr;
    proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    proxy_set_header X-Forwarded-Proto $scheme;
  }
  # 页面服务代理
  location /webfunny/ {
    proxy_pass http://x.x.x.x:webport/webfunny/;
    proxy_redirect default ;
  }
}

# CCE命令

# 查看xxx-sit空间下的所有pod
kubectl get pods -n xxx-sit
# 查看该pod在控制台的输出日志 
kubectl logs -f pod名称 -n xxx-sit 
# 进入该pod容器中
kubectl exec -it pod名称 sh -n xxx-sit 




# 查看该pod在控制台的输出日志
